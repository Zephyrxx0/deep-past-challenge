Create scripts/check_data_integrity.py with:

1. **Imports** (AGENTS.md order):
   - Standard: import argparse, import sys, from pathlib import Path
   - Third-party: import pandas as pd
   - Local: (none)

2. **Constants**:
   - DATA_DIR = Path('data')
   - REQUIRED_FILES = {
       'stage1_train': DATA_DIR / 'stage1_train.csv',
       'stage2_train': DATA_DIR / 'stage2_train.csv',
       'stage3_train': DATA_DIR / 'stage3_train.csv',
       'stage3_val': DATA_DIR / 'stage3_val.csv'
     }
   - REQUIRED_COLUMNS = ['transliteration_normalized', 'translation_normalized']
   - COMPETITION_TEST = DATA_DIR / 'competition' / 'test.csv'  # if exists, should not overlap with train

3. **Function check_provenance(data_dir: Path = DATA_DIR) -> bool:**
   - """Validate all required stage CSVs exist with required columns."""
   - for stage_name, filepath in REQUIRED_FILES.items():
     - if not filepath.exists():
       - raise FileNotFoundError(f"Dataset not found: {filepath}. Run data preprocessing first.")
     - df = pd.read_csv(filepath, nrows=5)  # Read only header + few rows for efficiency
     - missing_cols = [col for col in REQUIRED_COLUMNS if col not in df.columns]
     - if missing_cols:
       - raise ValueError(f"Dataset {filepath} missing required columns: {missing_cols}")
   - return True

4. **Function check_split_integrity(data_dir: Path = DATA_DIR) -> bool:**
   - """Validate no overlap between validation and training splits."""
   - # Rule 1: stage3_val must not overlap with stage1_train or stage2_train (prevents indirect leakage)
   - stage1_df = pd.read_csv(REQUIRED_FILES['stage1_train'])[REQUIRED_COLUMNS]
   - stage2_df = pd.read_csv(REQUIRED_FILES['stage2_train'])[REQUIRED_COLUMNS]
   - stage3_val_df = pd.read_csv(REQUIRED_FILES['stage3_val'])[REQUIRED_COLUMNS]
   - 
   - # Check overlap using transliteration as key (normalized form)
   - stage1_translit = set(stage1_df['transliteration_normalized'].dropna())
   - stage2_translit = set(stage2_df['transliteration_normalized'].dropna())
   - stage3_val_translit = set(stage3_val_df['transliteration_normalized'].dropna())
   - 
   - overlap_stage1 = stage3_val_translit.intersection(stage1_translit)
   - overlap_stage2 = stage3_val_translit.intersection(stage2_translit)
   - 
   - if overlap_stage1:
     - raise ValueError(f"Split integrity violation: {len(overlap_stage1)} validation examples found in stage1_train. Example: {list(overlap_stage1)[:3]}")
   - if overlap_stage2:
     - raise ValueError(f"Split integrity violation: {len(overlap_stage2)} validation examples found in stage2_train. Example: {list(overlap_stage2)[:3]}")
   - 
   - # Rule 2: If competition/test.csv exists, stage2_train and stage3_train should not overlap with it
   - if COMPETITION_TEST.exists():
     - test_df = pd.read_csv(COMPETITION_TEST)
     - if 'transliteration' in test_df.columns:  # May not have _normalized yet
       - test_translit = set(test_df['transliteration'].dropna())
       - stage3_train_df = pd.read_csv(REQUIRED_FILES['stage3_train'])[REQUIRED_COLUMNS]
       - stage3_train_translit = set(stage3_train_df['transliteration_normalized'].dropna())
       - overlap_test = stage3_train_translit.intersection(test_translit)
       - if overlap_test:
         - raise ValueError(f"Test set contamination: {len(overlap_test)} test examples found in stage3_train.")
   - 
   - return True

5. **Function main()**:
   - ArgumentParser with --check (choices=['all', 'provenance', 'splits'], default='all')
   - Parse args
   - try:
     - if args.check in ['all', 'provenance']:
       - check_provenance()
       - print("✓ Provenance check passed: all required datasets exist with correct columns")
     - if args.check in ['all', 'splits']:
       - check_split_integrity()
       - print("✓ Split integrity check passed: no train/validation overlap detected")
     - print(f"\n✓ Data integrity validation complete: {args.check}")
     - sys.exit(0)
   - except (FileNotFoundError, ValueError) as e:
     - print(f"✗ Data integrity check failed: {e}", file=sys.stderr)
     - sys.exit(1)

6. **Entry point**: if __name__ == '__main__': main()
